DROP TABLE T_QUARTO;
DROP TABLE T_RESERVA;

CREATE TABLE T_QUARTO(
NR_QUARTO       INT     PRIMARY KEY,
CATEGORIA       INT     NOT NULL,
MAXPESSOAS      INT     NOT NULL,
VLRDIARIA       FLOAT   NOT NULL
)

CREATE TABLE T_RESERVA(
ID_RESERVA      INT     PRIMARY KEY,
QUARTO_RESERVA  INT     NOT NULL,
DTENTRADA       DATE    NOT NULL,
DTSAIDA         DATE    NOT NULL,
QTDPESSOAS      INT     NOT NULL
)

CREATE SEQUENCE SQ_RESERVA
 START WITH     1
 INCREMENT BY   1;
